pipeline {
    agent any  

    stages {
        stage('Checkout') {
            steps {
                echo "Hello Checkout"
                git 'https://github.com/Raja13196/git-tutorial.git'
                script {
                    checkout scm
                    // OR, to use retry
                    checkout retry: 5, scm
                }
                
            }
        }
        stage('Deploy') {
            steps {
                echo 'Hello deploy'
                bat 'cd demo/demo'
                bat 'mvn clean'
            }
        }
        stage('Test') {
            steps {
                echo 'Hello test'
            }
        }
    }
}
